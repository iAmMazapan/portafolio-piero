// BFS para encontrar el camino m√°s corto en grafos no ponderados
    #include <bits/stdc++.h>
    using namespace std;

    vector<int> bfs(vector<vector<int>>& graph, int start, int end) {
        int n = graph.size();
        vector<int> dist(n, -1);
        vector<int> parent(n, -1);
        queue<int> q;
        
        dist[start] = 0;
        q.push(start);
        
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            
            if (u == end) break;
            
            for (int v : graph[u]) {
                if (dist[v] == -1) {
                    dist[v] = dist[u] + 1;
                    parent[v] = u;
                    q.push(v);
                }
            }
        }
        
        // Reconstruir camino
        vector<int> path;
        if (dist[end] == -1) return path; // No existe camino
        
        for (int v = end; v != -1; v = parent[v]) {
            path.push_back(v);
        }
        reverse(path.begin(), path.end());
        
        return path;
    }

    // Complejidad: O(V + E)
    // Uso: grafos no ponderados, nivel m√°s corto

    typedef pair<long long, int> pli;

    vector<long long> dijkstra(vector<vector<pli>>& graph, int start) {
        int n = graph.size();
        vector<long long> dist(n, LLONG_MAX);
        priority_queue<pli, vector<pli>, greater<pli>> pq;
        
        dist[start] = 0;
        pq.push({0, start});
        
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            
            if (d > dist[u]) continue;
            
            for (auto [v, weight] : graph[u]) {
                if (dist[u] + weight < dist[v]) {
                    dist[v] = dist[u] + weight;
                    pq.push({dist[v], v});
                }
            }
        }
        
        return dist;
    }

// DSU con compresi√≥n de caminos y union por rango
    #include <bits/stdc++.h>
    using namespace std;

    class DSU {
    private:
        vector<int> parent, rank;
        
    public:
        DSU(int n) : parent(n), rank(n, 0) {
            iota(parent.begin(), parent.end(), 0);
        }
        
        int find(int x) {
            if (parent[x] != x) {
                parent[x] = find(parent[x]); // Compresi√≥n de caminos
            }
            return parent[x];
        }
        
        bool unite(int x, int y) {
            int px = find(x), py = find(y);
            if (px == py) return false;
            
            // Union por rango
            if (rank[px] < rank[py]) swap(px, py);
            parent[py] = px;
            if (rank[px] == rank[py]) rank[px]++;
            
            return true;
        }
        
        bool connected(int x, int y) {
            return find(x) == find(y);
        }
    };
